def getgeo(postcode):
    lat_lng_coords = None
    i = 0
    while(lat_lng_coords is None and i < 10):
        g = geocoder.google('{}, Toronto, Ontario'.format(postcode))
        lat_lng_coords = g.latlng
        print(lat_lng_coords)
        i += 1
    return lat_lng_coords
